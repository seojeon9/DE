# 우분투 기반 컨테이너 실행
docker run -it --name de_base ubuntu:20.04
# -it : 컨테이너의 터미널에 접근
# --name : 컨테이너 이름

# 패키지 관리자 업데이트
apt update -y
# -y : 업데이트 진행여부를 물어볼 때 y를 자동으로 입력
apt upgrade -y

apt install wget vim unzip sudo -y
# wget : 웹에서 파일을 다운로드
# vim : 텍스트 에디터
# unzip : 알집
# sudo : sudo 명령어를 쓰기위해 설치


사용자 생성
사용자@호스트:위치[#,$]
	사용자 : 사용자
	호스트 : 네트워크에서 해당 컴퓨터를 식별하기 위한 이름
	# : super user
	$ : 일반사용자

# 사용자 생성
useradd -m big
-m : 사용자를 생성하면서 홈디렉터리도 같이 생성

# 사용자 목록 확인
cat /etc/passwd
big:x:1000:1000::/home/big:/bin/sh
# 1. 사용자명
# 2. 암호화된 password
# 3. userId
# 4. 그룹번호
# 5. 추가정보
# 6. 홈디랙터리
# 7. login 시 사용할 쉘

# big 사용자의 쉘 설정을 변경
big:x:1000:1000::/home/big:/bin/sh => big:x:1000:1000::/home/big:/bin/bash
# /etc/passwd 파일을 수정
vim /etc/passwd
#i를 사용해 입력모드로 진입
#수정
esc + wq (저장 및 나가기)

#사용자 상세 정보
vim /etc/shadow

#root권한 빌려오기
sudo 명령어

# 사용자 비밀번호 설정
passwd root
passwd big

# sudoers 파일 수정
vim /etc/sudoers
big     ALL=(ALL:ALL) ALL
